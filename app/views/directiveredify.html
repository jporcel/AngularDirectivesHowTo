Simple DOM manipulation

A very simple "attribute" directive that changes the text color of the associated element to red...

app.directive('redify', function() {
  return {
    restrict: 'A',
    link: function(scope, element, attrs) {
      // change the color to red
      element.css('color', 'red');
    }
  };
});

<p redify>hello world</p>
