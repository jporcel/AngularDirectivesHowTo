app.directive('mediumYellowRectangle', function() {
  return {
    restrict: 'AE',
    transclude: true,
    template: "<div style='width:120px;margin:10px;height:80px;background-color:yellow;'><div ng-transclude></div></div>",
    link: function(scope, element, attrs) {
            
    }
  };
});
