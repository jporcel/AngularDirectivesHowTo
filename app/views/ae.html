Make a directive that can be used both as an attribute or an element.

Example: Big green circle directive:

app.directive('bigGreenCircle', function() {
  return {
    restrict: 'AE',    
    link: function(scope, element, attrs) {
        
        element.width(200);
        element.height(200);
        element.css('background-color', 'lime');
        element.css('border-radius', 125);
        element.css('display', 'inline-block');
    }
  };
});

<div big-green-circle>hello</div>
hello
<big-green-circle>goodbye</big-green-circle>
goodbye

Warning

A template in a directive makes sense for element directive, not always for a attribute directive. it will replace the content of the associated html object

Example, a "small-red-square" directive:

app.directive('smallRedSquare', function() {
  return {
    restrict: 'AE',
    template: "<div style='width:100px;height:100px;background-color:red;'></div>",
    link: function(scope, element, attrs) {
            
    }
  };
});

Using it as an element:

<small-red-square></small-red-square>

Using it as an attribute:

<p small-red-square>HELLO WORLD</small-red-quare>

HELLO WORLD

The text is gone, the content of the p tag has been replaced with the template of the directive. The dom tree looks like: 
<p>
    <div>this element is the template directive...</div>
</p>
