Using an Angular service in a directive

A stand-alone next button directive control that interacts with the angular $location service...

JS

app.directive('nextButton', ['$location', function($location) {
    var myDir = {
        restrict: 'E',
        template: '<a class="next-button button">Next <i class="icon-large icon-circle-arrow-right"><'/i></a>',
        link: function(scope, element, attrs) {
            element.bind('click', function() {
                $location.path(attrs.href);
                scope.$apply() 
            });
        }
    };
    return myDir;
}]);

HTML

<next-button href="adipoli" ></next-button>
