Pass arguments to directive I

Easiest method: 3rd argument of the link function...

Pass an argument to the directive. This directive is similar to the redify directive except you can now specify which color you want to use...

app.directive('colorify', function() {
  return {
    restrict: 'A',
    link: function(scope, element, attrs) {
      // change the color to whatever was passed in attribute
      element.css('color', attrs.colorify);
    }
  };
});

<p colorify="lime">hello world</p>
